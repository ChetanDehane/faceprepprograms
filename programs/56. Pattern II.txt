Pattern II

Write a program to print the following pattern.

Sample Input:

5

Sample Output:

1
3*2
4*5*6
10*9*8*7
11*12*13*14*15

program:

#include <iostream> 
using namespace std; 
int main()
{ 
	int n,j, k = 0; 
	cin>>n;
	for (int i=1; i<=n; i++) 
	{ 
		if (i%2 != 0) 
		{ 
			for (j=k+1; j<k+i; j++) 
				cout << j << "*"; 
			cout << j++ << endl; 
			k = j;	 
		} 
		else
		{ 
			k = k+i-1; 
			for (j=k; j>k-i+1; j--) 
				cout << j << "*"; 
			cout << j << endl;	 
		} 
	} 

	return 0; 
}
